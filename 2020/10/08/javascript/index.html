8.函数

//第一种方式
function abs(x) {
    if(x>0){
        return x;
    }else {
        return -x;
    }
}
//第二种方式
var abs=function (x) {
    if(x>0){
        return x;
    }else {
        return -x;
    }
}
----------------------------------------------------------
    //arguments是JS免费赠送的关键字，代表传递进来的所有参数，是一个数组

        console.log(x);
    }
---------------------------------------------------------------
//ES6新特性   rest
    var abs=function (x,...rest) {
        if(x>0){
            console.log(x);
        }else {
            console.log(-x);
        }
        console.log(rest); //必须放在最后，打印的是多余的参数，作为数组
    }

3.JSON

  • 是一种轻量级的数据交换格式
  • 层次结构简洁清晰，理想的数据交换语言
  • 易于编写，同时也易于机器解析和生成，能有效的提升网络传输效率

在js中，一些皆对象，任何js支持的类型都能用JSON表示

格式：

  • 对象 {}
  • 数组[]
  • 所有的键值对 K:V
//JSON字符串和JS对象的转换
let person = {
    name: "hs",
    age: 18,
    email: "1972728214@qq.com",
    score: 0
};
//将对象转换为JSON的格式
JSON.stringify(person);
//解析JSON数据    参数为字符串
JSON.parse('{"name":"hs","age":18,"email":"1972728214@qq.com","score":0}')

2.操作DOM对象

DOM：文档对象模型

浏览器就是一个DOM树形结构

  • 更新：更新DOM节点
  • 遍历DOM节点，得到DOM节点
  • 删除：删除DOM节点
  • 添加一个新的DOM节点

操作DOM节点必须先获得DOM节点；

获得DOM节点

<body>
    <h1>标题1</h1>
    <p id="p1"></p>
    <p class="p2"></p>
    <div class="father"></div>
    <script>
        //对应css选择器
        document.getElementsByClassName('p2');  //类
        document.getElementById('p1');          //id
        document.getElementsByTagName('h1')      //标签
        var father=document.getElementById('father'); //注意：id用于获取父节点下的子属性
        father.children //获取父节点下的所有子节点

    </script>
</body>

更新DOM节点

// 先获取DOM节点
 var father=document.getElementById('father');
// 操作文本
father.innertext=''
father.innerHTML='<strong></strong>'
//操作JS
father.style.color=''  // 属性是字符串
father.style.padding=''

删除节点

步骤：先获取父节点，再通过父节点删除自己

var father=document.getElementById('father');
//删除子节点
father.removeChild()
//注意：动态删除，节点时刻变化
